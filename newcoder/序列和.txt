#include<stdio.h>
#include<stdlib.h>

int findit(int n, int L, int* c,int* length){
    int i,len=L;
    while(len<=100){
        i=0;
        while(i<=n/2){
            if ( ((i+(i+len-1))*len)/2 == n){
                *c=i;
                *length=len;
                return 0; //found
            }
            i++;
        }
        len++;
    }
    return 1;
}
int main(){
    int i,l=0,n,L,index=-1;
    scanf("%d %d",&n,&L);
    if(!(n<=10e9&&n>=1)||!(L>=2&&L<=100)){
        printf("Wrong parameters!\n");
        exit(-1);
    }
    if(findit(n,L,&index,&l)){
        printf("NO");
    }
    else{
        for(i=0;i<l;i++){
            if(i==l-1)
                printf("%d",(index+i));
            else
                printf("%d ",index+i);
        }
    }
    return 0;
}
